680  void initialize(bool read_only)
681  {

683  lock_ = std::make_shared<SingleUnlock>(planning_scene_monitor_.get(), read_only);
684  }
685 
686  MOVEIT_STRUCT_FORWARD(SingleUnlock);
687 
688  // we use this struct so that lock/unlock are called only once
689  // even if the LockedPlanningScene instance is copied around
690  struct SingleUnlock
691  {
692  SingleUnlock(PlanningSceneMonitor* planning_scene_monitor, bool read_only)
693  : planning_scene_monitor_(planning_scene_monitor), read_only_(read_only)
694  {
695  if (read_only)
696  planning_scene_monitor_->lockSceneRead();
697  else
698  planning_scene_monitor_->lockSceneWrite();
699  }
700  ~SingleUnlock()
701  {
702  if (read_only_)
703  planning_scene_monitor_->unlockSceneRead();
704  else
705  planning_scene_monitor_->unlockSceneWrite();
706  }
707  PlanningSceneMonitor* planning_scene_monitor_;
708  bool read_only_;
709  };
710 
711  PlanningSceneMonitorPtr planning_scene_monitor_;
712  SingleUnlockPtr lock_;
713 };
